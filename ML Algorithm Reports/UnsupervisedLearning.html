cluster analysis using generated data

set.seed(101)
x=matrix(rnorm(100*2),ncol = 2,nrow = 100)
addMean=matrix(rnorm(8,sd = 4),4,2)
index=sample(1:4,100,replace = TRUE)
x=x+addMean[index,]
plot(x,col=index,pch=20,cex=2)
